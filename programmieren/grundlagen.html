Beispiel in C# (Konsolenanwendung)

using System;


public class Testklasse
{

    public void meine_zahl(int zahl)
    {
        Console.WriteLine("Ihre Zahl ist: " + zahl);
    }

    public void mein_text(string text)
    {
        Console.WriteLine("Ihr Wort ist: " + text);
    }

    public void bedingung(int zahl)
    {
        int i = zahl;

        if (i == 10)
        {
            Console.WriteLine(i + " ist genau gleich 10");
        }
        else if (i > 10)
        {
            Console.WriteLine(i + " ist größer als 10");
        }
        else 
        {
            Console.WriteLine(i + " ist kleiner als 10");
        }
    }

    public void schleife(int zahl)
    {
        Console.WriteLine("Die Zahlen von 1 bis " + zahl + ":");

        for (int i = 1; i <= zahl; i++)
        {            
            Console.WriteLine(i);
        }
    }

    public void kopfschleife(int zahl)
    {
        Console.WriteLine("Kopfgesteuerte Schleife, solange die eingegebene Zahl kleiner als 10 ist, höherzählen:");

        while (zahl < 10)
        {            
            Console.WriteLine(zahl);
            zahl++;
        }
    }

    public void fussschleife(int zahl)
    {
        Console.WriteLine("Fußgesteuerte Schleife, solange die eingegebene Zahl kleiner als 10 ist, höherzählen:");

        do
        {            
            Console.WriteLine(zahl);
            zahl++;
        }
        while (zahl < 10);
    }

}



class Program
{
    static void Main()
    {
        Testklasse test = new Testklasse();
        test.meine_zahl(10);
        test.mein_text("Na sowas");
        test.bedingung(9);
        test.schleife(7);
        test.kopfschleife(5);
        test.fussschleife(3);
    }
    
}

class Testklasse{
    public function bedingung($zahl){
        if($zahl == 10){
            echo "Die Zahl " . $zahl . " ist genau gleich 10<br>";
        }
        elseif($zahl > 10){
            echo "Die Zahl " . $zahl . " ist größer als 10<br>";
        }
        else{
            echo "Die Zahl " . $zahl . " ist nicht größer als 10<br>";
        }
    }

    public function schleife($zahl){
        echo "Schleife von 1 bis " . $zahl . "<br>";
        for($i = 1; $i < $zahl; $i++){ 
            echo $i . "<br>";
        }
    }

    public function kopfschleife($zahl){
        echo "Kopfgesteuerte Schleife von 1 bis " . $zahl . "<br>";
        
        $i = 1;

        while($i <= $zahl){
            echo $i . "<br>";
            $i++;
        }
    }

    public function fussschleife($zahl){
        echo "Kopfgesteuerte Schleife von 1 bis " . $zahl . "<br>";

        $i = 1;

        do{
            echo $i . "<br>";
            $i++;
        }
        while($i <= $zahl);
    }
}
